/* --- PLAYER (inertia-based) --- */ let player = { x: 0, y: 0, vx: 0, vy: 0, size: 20, accel: 0.4, friction: 0.85, maxSpeed: 4 }; let keys = {}; let running = false; /* --- INPUT --- */ document.addEventListener("keydown", e => keys[e.key] = true); document.addEventListener("keyup", e => keys[e.key] = false); /* --- WORLD --- */ const world = { width: 3000, height: 3000 }; /* --- CAMERA --- */ let camera = { x: 0, y: 0 }; /* --- UPDATE --- */ function update() { if (!running) return; let ax = 0; let ay = 0; if (keys["w"] || keys["ArrowUp"]) ay -= player.accel; if (keys["s"] || keys["ArrowDown"]) ay += player.accel; if (keys["a"] || keys["ArrowLeft"]) ax -= player.accel; if (keys["d"] || keys["ArrowRight"]) ax += player.accel; // apply acceleration player.vx += ax; player.vy += ay; // clamp speed const speed = Math.sqrt(player.vx * player.vx + player.vy * player.vy); if (speed > player.maxSpeed) { player.vx = (player.vx / speed) * player.maxSpeed; player.vy = (player.vy / speed) * player.maxSpeed; } // apply friction player.vx *= player.friction; player.vy *= player.friction; // move player.x += player.vx; player.y += player.vy; // bounds player.x = Math.max(0, Math.min(world.width, player.x)); player.y = Math.max(0, Math.min(world.height, player.y)); // camera follows player (smooth) camera.x += (player.x - camera.x - canvas.width / 2) * 0.1; camera.y += (player.y - camera.y - canvas.height / 2) * 0.1; } /* --- DRAW --- */ function draw() { // background ctx.fillStyle = "#555"; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.save(); // move world relative to camera ctx.translate(-camera.x, -camera.y); /* --- GROUND (subtle texture, no grid lines) --- */ for (let x = 0; x < world.width; x += 100) { for (let y = 0; y < world.height; y += 100) { ctx.fillStyle = (x + y) % 200 === 0 ? "#5a5a5a" : "#585858"; ctx.fillRect(x, y, 100, 100); } } /* --- PLAYER --- */ ctx.fillStyle = "white"; ctx.beginPath(); ctx.arc(player.x, player.y, player.size, 0, Math.PI * 2); ctx.fill(); ctx.restore(); } /* --- LOOP (keep your fixed timestep) --- */ let lastTime = 0; const fps = 60; const frameDuration = 1000 / fps; let accumulator = 0; function loop(currentTime) { if (!running) return; const delta = currentTime - lastTime; lastTime = currentTime; accumulator += delta; while (accumulator >= frameDuration) { update(); accumulator -= frameDuration; } draw(); requestAnimationFrame(loop); }
